149  public static IMod LoadMod(IMod mod)
150  {
151  if (mod == null) return null;
152  var modType = mod.GetType();
153  var attrib = modType.GetFirstAttribute<ALoadOrderAttribute>();
154  string name = mod.GetCleanId();
155  if (attrib != null)
156  {
157  foreach (var pair in ModIdReliance.ToArray())
158  {
159  var i = pair.Value.FindIndex(m => m.GetCleanId() == name);
160  if (i != -1)
161  ModIdReliance[pair.Key][i] = mod;
162  }
163  foreach (var id in attrib.beforeIds)
164  {
165  if (!ModIdReliance.ContainsKey(id.GetCleanId()))
166  ModIdReliance.Add(id, new List<IMod>());
167  if (!ModIdReliance[id].Contains(mod))
168  ModIdReliance[id].Add(mod);
169  }
170  foreach (var id in attrib.afterIds)
171  {
172  if (!ModIdReliance.ContainsKey(name))
173  ModIdReliance.Add(name, new List<IMod>());
174  if (ModIdReliance[name].FindIndex(m => m.GetCleanId() == id.GetCleanId()) != -1)
175  ModIdReliance[name].Add(new Placeholder(id.GetCleanId()));
176  }
177  if (attrib.beforeIds.Count > 0)
178  return mod;
179  }
180  try
181  {
182  if (!Pathfinder.IsModIdentifierValid(name, true))
183  return null; // never reached due to throw
184  Logger.Info("Loading mod '{0}'", name);
185  CurrentMod = mod;
186  mod.Load();
187  UnloadedModIds.Remove(name);
188  LoadedMods.Add(name, mod);
189  GUI.ModOptions.Handler.LoadFor(mod);
190  if (ModIdReliance.ContainsKey(name))
191  foreach (var internalMod in ModIdReliance[name])
192  LoadMod(internalMod);
193  }
194  catch (Exception ex)
195  {
196  Logger.Error("Mod '{0}' of file '{1}' failed to load:\n\t{2}", modType.FullName, Path.GetFileName(modType.Assembly.Location), ex);
197  UnloadMod(mod);
198  UnloadedModIds.Remove(name);
199  CurrentMod = null;
200  return null;
201  }
202  CurrentMod = null;
203  return mod;
204  }
